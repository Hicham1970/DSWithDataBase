Sub HideAll()

'Hide Horizontal scroll bar
ActiveWindow.DisplayHorizontalScrollBar = False

' hide vertical scroll bar
ActiveWindow.DisplayVerticalScrollBar = False

'hide the Rows/columns heading
ActiveWindow.DisplayHeadings = False

'hide worksheet tab
ActiveWindow.DisplayWorkbookTabs = False

'hide status Bar(bottom of the window
Application.DisplayStatusBar = False

'hide formulaBar
Application.DisplayFormulaBar = False

' hide rubbon menu et le quick acces toolBar
Application.ExecuteExcel4Macro "show.toolbar(""Ribbon"",false)"

'Display FullScreen
Application.DisplayFullScreen = True

End Sub

Sub ShowAll()

'Exit FullScreen
Application.DisplayFullScreen = False

'Show Horizontal scroll bar
ActiveWindow.DisplayHorizontalScrollBar = True

' Show vertical scroll bar
ActiveWindow.DisplayVerticalScrollBar = True

'Show the Rows/columns heading
ActiveWindow.DisplayHeadings = True

'Show worksheet tab
ActiveWindow.DisplayWorkbookTabs = True

'Show status Bar(bottom of the window
Application.DisplayStatusBar = True

'Show formulaBar
Application.DisplayFormulaBar = True


' Show rubbon menu et le quick acces toolBar
Application.ExecuteExcel4Macro "show.toolbar(""Ribbon"", True)"

End Sub
